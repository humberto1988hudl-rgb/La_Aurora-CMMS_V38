function updateGauges(temp, hum) {
  tempGauge.value = temp;
  humGauge.value = hum;
}

// ============= MQTT ==============
const client = mqtt.connect('ws://192.168.1.11:9001');
client.on('connect', () => client.subscribe('cuarto1/#'));

document.getElementById("lightM1").style.backgroundColor = "red";
document.getElementById("lightM2").style.backgroundColor = "red";
document.getElementById("lightM3").style.backgroundColor = "red";

let machineTimers = {
  maquina1: null,
  maquina2: null,
  maquina3: null
};

function resetMachineTimer(maquina, lightId) {
  if (machineTimers[maquina]) clearTimeout(machineTimers[maquina]);
  machineTimers[maquina] = setTimeout(() => {
    document.getElementById(lightId).style.backgroundColor = "red";
  }, 3000);
}

client.on('message', (topic, message) => {
  const data = JSON.parse(message.toString());
  if (topic === 'cuarto1/tempHum') updateGauges(data.temp, data.hum);

  if (topic.includes("maquina1")) {
    updateHorometro("maquina1", data);
    document.getElementById("lightM1").style.backgroundColor = data.activo ? "green" : "red";
    resetMachineTimer("maquina1", "lightM1");
  } else if (topic.includes("maquina2")) {
    updateHorometro("maquina2", data);
    document.getElementById("lightM2").style.backgroundColor = data.activo ? "green" : "red";
    resetMachineTimer("maquina2", "lightM2");
  } else if (topic.includes("maquina3")) {
    updateHorometro("maquina3", data);
    document.getElementById("lightM3").style.backgroundColor = data.activo ? "green" : "red";
    resetMachineTimer("maquina3", "lightM3");
  }
});

function updateHorometro(maquina, data) {
  const selected = document.getElementById("machineSelect").value;
  if ((maquina === "maquina1" && selected === "RSC-100") ||
      (maquina === "maquina2" && selected === "TPC-100") ||
      (maquina === "maquina3" && selected === "TRF-100")) {
    document.getElementById("horometroValue").innerText = `${data.h}h ${data.m}m ${data.s}s`;
  }
}
